### 第 2 步：第一个 Demo - 智能对焦区域选择 (1 小时)

#### 目标
使用 YOLOv8 检测画面中的人脸，自动选择对焦区域。

#### 代码实现
创建 `smart_autofocus_demo.py`:

```python
import cv2
from ultralytics import YOLO
import numpy as np

class SmartAutofocus:
    def __init__(self):
        # 加载 YOLOv8 模型（nano 版本，速度快）
        self.model = YOLO('yolov8n.pt')
        
        # 定义优先级（类别 ID -> 优先级）
        self.priority = {
            0: 10,   # person
            15: 8,   # cat
            16: 8,   # dog
            1: 5,    # bicycle
            2: 5,    # car
        }
    
    def detect_objects(self, frame):
        """检测画面中的对象"""
        results = self.model(frame, verbose=False)
        return results[0]
    
    def select_focus_region(self, frame, results):
        """选择最重要的对焦区域"""
        if len(results.boxes) == 0:
            # 没有检测到对象，使用画面中心
            h, w = frame.shape[:2]
            return (w//2, h//2)
        
        # 根据优先级和距离选择
        best_box = None
        best_score = -1
        
        for box in results.boxes:
            cls_id = int(box.cls[0])
            conf = float(box.conf[0])
            
            # 计算优先级分数
            priority_score = self.priority.get(cls_id, 1)
            
            # 考虑置信度和面积
            x1, y1, x2, y2 = box.xyxy[0]
            area = (x2 - x1) * (y2 - y1)
            
            # 综合分数
            score = priority_score * conf * (area / (frame.shape[0] * frame.shape[1]))
            
            if score > best_score:
                best_score = score
                best_box = box
        
        # 返回中心点
        if best_box is not None:
            x1, y1, x2, y2 = best_box.xyxy[0]
            center_x = int((x1 + x2) / 2)
            center_y = int((y1 + y2) / 2)
            return (center_x, center_y)
        
        # 默认中心
        h, w = frame.shape[:2]
        return (w//2, h//2)
    
    def compute_sharpness(self, image):
        """计算图像清晰度（拉普拉斯方差）"""
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        return cv2.Laplacian(gray, cv2.CV_64F).var()
    
    def draw_focus_point(self, frame, point):
        """在画面上绘制对焦点"""
        x, y = point
        cv2.circle(frame, (x, y), 20, (0, 255, 0), 2)
        cv2.circle(frame, (x, y), 5, (0, 255, 0), -1)
        cv2.putText(frame, "Focus", (x+25, y), 
                    cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 0), 2)
        return frame
    
    def run(self, source=0):
        """运行智能对焦 Demo"""
        cap = cv2.VideoCapture(source)
        
        if not cap.isOpened():
            print("无法打开视频源")
            return
        
        print("按 'q' 退出")
        
        while True:
            ret, frame = cap.read()
            if not ret:
                break
            
            # 检测对象
            results = self.detect_objects(frame)
            
            # 选择对焦区域
            focus_point = self.select_focus_region(frame, results)
            
            # 计算清晰度
            sharpness = self.compute_sharpness(frame)
            
            # 绘制对焦点
            frame = self.draw_focus_point(frame, focus_point)
            
            # 显示清晰度
            cv2.putText(frame, f"Sharpness: {sharpness:.1f}", (10, 30),
                       cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 255, 255), 2)
            
            cv2.imshow('Smart Autofocus', frame)
            
            if cv2.waitKey(1) & 0xFF == ord('q'):
                break
        
        cap.release()
        cv2.destroyAllWindows()

if __name__ == "__main__":
    af = SmartAutofocus()
    af.run(source=0)  # 0 为摄像头，或传入视频文件路径
```

### 第 3 步：强化学习对焦环境 (2 小时)

#### 目标
创建一个简单的强化学习对焦环境，训练智能体学习对焦策略。

#### 实现对焦环境
创建 `autofocus_env.py`:

```python
import gymnasium as gym
import numpy as np
import cv2
from gymnasium import spaces

class SimpleAutofocusEnv(gym.Env):
    """简单的对焦环境"""
    
    def __init__(self):
        super().__init__()
        
        # 动作空间：0=后退，1=停止，2=前进
        self.action_space = spaces.Discrete(3)
        
        # 观察空间：简化为清晰度分数 + 当前对焦位置
        self.observation_space = spaces.Box(
            low=np.array([0, 0]),
            high=np.array([1000, 100]),
            dtype=np.float32
        )
        
        # 模拟参数
        self.focus_range = 100  # 对焦范围
        self.optimal_focus = np.random.randint(20, 80)  # 随机最优位置
        self.current_focus = 50  # 当前对焦位置
        self.max_steps = 20  # 最大步数
        self.current_step = 0
        
    def step(self, action):
        """执行一步动作"""
        # 更新对焦位置
        if action == 0:  # 后退
            self.current_focus = max(0, self.current_focus - 5)
        elif action == 2:  # 前进
            self.current_focus = min(100, self.current_focus + 5)
        # action == 1 停止，位置不变
        
        # 计算清晰度（模拟：越接近最优位置，清晰度越高）
        distance = abs(self.current_focus - self.optimal_focus)
        sharpness = max(0, 1000 * (1 - distance / 50))
        
        # 计算奖励
        reward = sharpness / 1000  # 归一化到 [0, 1]
        
        # 惩罚过多步数
        reward -= 0.01
        
        # 如果接近最优位置，额外奖励
        if distance < 5:
            reward += 1.0
        
        # 检查是否结束
        self.current_step += 1
        done = (self.current_step >= self.max_steps) or (distance < 3)
        
        # 观察状态
        observation = np.array([sharpness, self.current_focus], dtype=np.float32)
        
        info = {
            'sharpness': sharpness,
            'distance': distance,
            'optimal': self.optimal_focus
        }
        
        return observation, reward, done, False, info
    
    def reset(self, seed=None, options=None):
        """重置环境"""
        super().reset(seed=seed)
        
        # 随机化最优对焦位置
        self.optimal_focus = np.random.randint(20, 80)
        self.current_focus = np.random.randint(10, 90)
        self.current_step = 0
        
        # 初始清晰度
        distance = abs(self.current_focus - self.optimal_focus)
        sharpness = max(0, 1000 * (1 - distance / 50))
        
        observation = np.array([sharpness, self.current_focus], dtype=np.float32)
        
        return observation, {}
    
    def render(self, mode='human'):
        """可视化（可选）"""
        print(f"Step {self.current_step}: Focus={self.current_focus:.1f}, "
              f"Optimal={self.optimal_focus}, "
              f"Distance={abs(self.current_focus - self.optimal_focus):.1f}")


# 测试环境
if __name__ == "__main__":
    env = SimpleAutofocusEnv()
    obs, info = env.reset()
    
    print("初始状态:", obs)
    
    for i in range(10):
        action = env.action_space.sample()  # 随机动作
        obs, reward, done, truncated, info = env.step(action)
        env.render()
        
        if done:
            print("完成！")
            break
    
    env.close()
```

#### 训练强化学习智能体
创建 `train_autofocus_rl.py`:

```python
from stable_baselines3 import PPO
from stable_baselines3.common.env_util import make_vec_env
from autofocus_env import SimpleAutofocusEnv

# 创建环境
env = make_vec_env(lambda: SimpleAutofocusEnv(), n_envs=4)

# 创建模型
model = PPO(
    "MlpPolicy",
    env,
    learning_rate=3e-4,
    n_steps=2048,
    batch_size=64,
    n_epochs=10,
    verbose=1
)

# 训练
print("开始训练...")
model.learn(total_timesteps=50000)
print("训练完成！")

# 保存模型
model.save("autofocus_ppo")

# 测试训练好的模型
print("\n测试模型...")
test_env = SimpleAutofocusEnv()
obs, info = test_env.reset()

for i in range(20):
    action, _states = model.predict(obs, deterministic=True)
    obs, reward, done, truncated, info = test_env.step(action)
    
    print(f"Step {i}: Action={action}, Reward={reward:.3f}, "
          f"Distance={info['distance']:.1f}")
    
    if done:
        print("成功对焦！")
        break

test_env.close()
```
